
def count_chars(string):
    uppercase_count = 0
    lowercase_count = 0
    whitespace_count = 0
    
    for char in string:
        if char.isupper():
            uppercase_count += 1
        elif char.islower():
            lowercase_count += 1
        elif char.isspace():
            whitespace_count += 1
    
    print("Uppercase:", uppercase_count)
    print("Lowercase:", lowercase_count)
    print("Whitespaces:", whitespace_count)

user_input = input("Enter any string: ")
count_chars(user_input)
